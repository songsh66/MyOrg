a=5,b=7,c=3

fun函数里面
a=5&&&&&;b=7&&&&&;c=3&&&&&;
temp=a;    //temp=5&&&&&;
a=b;       //a=7&&&&&;
b=temp;    //b=5&&&&&;
这个时候a=7&&&&&;b=5&&&&&;c=3&&&&&;
*temp=*b;  //*b=5,*temp=5;这个地方有个陷阱，temp在上面的代码里面已经变成了5&&&&&，和b是一样的
*b=*c;     //*b=*c=3，然后*temp=*（5&&&&&）=3；
*c=*temp;  //*temp=*（5&&&&&）=3；*c=*(3&&&&&)=3
